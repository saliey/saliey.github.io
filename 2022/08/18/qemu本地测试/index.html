qemu 是模拟 cpu 和硬件的虚拟机，所有在 x86 上运行的 arm 虚拟机都是基于 qemu

qemu-img create -f qcow2 android.img 15G
sudo qemu-system-x86_64 -m 2048 -boot d -enable-kvm -smp 3 -net nic -net user -accel hvf -hda android.img -cdrom

-accel hvf

qemu-img create -f qcow2 ubuntu20.qcow2 50G
qemu-system-x86_64
-machine type=q35,accel=hvf
-smp 6
-cpu host
-hda ubuntu20.qcow2
-m 16G
-vga virtio
-usb
-device usb-tablet
-display default,show-cursor=on -cdrom ubuntu-22.04.1-live-server-amd64.iso
